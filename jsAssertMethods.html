Node Assert Methods 

const assert = require('assert');

// Example objects for equality checks
const obj1 = {name: 'John', age: 30};
const obj2 = {name: 'John', age: 30};
const obj3 = {name: 'Jane', age: 25};

// Using assert.equal() for loose equality check
assert.equal(10, '10', 'Loose equality check failed'); // Passes
assert.equal(obj1, obj2, 'Loose equality check failed'); // Fails

// Using assert.strictEqual() for strict equality check
assert.strictEqual(10, '10', 'Strict equality check failed'); // Fails
assert.strictEqual(obj1, obj2, 'Strict equality check failed'); // Fails

// Using assert.deepEqual() for deep equality check
assert.deepEqual(10, '10', 'Deep equality check failed'); // Passes
assert.deepEqual(obj1, obj2, 'Deep equality check failed'); // Passes
assert.deepEqual(obj1, obj3, 'Deep equality check failed'); // Fails
